-- Database: sjs

CREATE DATABASE sjs
    WITH
    OWNER = admin
    ENCODING = 'UTF8'
    LC_COLLATE = 'C'
    LC_CTYPE = 'C'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;

CREATE TABLE IF NOT EXISTS account
(
    aid SERIAL PRIMARY KEY,
	  email VARCHAR,
	  pwd VARCHAR,
	  username VARCHAR
);

DROP TABLE chatlog;
CREATE TABLE IF NOT EXISTS chatlog
(
    cid SERIAL PRIMARY KEY,
	  senderId VARCHAR,
	  message VARCHAR,
	  label VARCHAR,
	  timestamp TIMESTAMP
);

CREATE TABLE IF NOT EXISTS Event_Header(
    eHdrId SERIAL PRIMARY KEY,
    eHdrTitle VARCHAR,
    eStatus VARCHAR,
    eHdrDesc VARCHAR,
    eHdrDate TIMESTAMP,
    eHdrCreateDate TIMESTAMP
);

CREATE TABLE IF NOT EXISTS Event_Detail(
    eDtlId SERIAL PRIMARY KEY,
    eDtlOptTitle VARCHAR,
    eDtlHdrId INTEGER,
    eDtlUsername VARCHAR,
    eDtlGameId VARCHAR,
    CONSTRAINT fk_EventDetail FOREIGN KEY (eDtlHdrId) REFERENCES Event_Header(eHdrId)
);

CREATE TABLE IF NOT EXISTS games
(
    gid SERIAL PRIMARY KEY,
	gname VARCHAR,
    gname_zh VARCHAR
);

CREATE TABLE IF NOT EXISTS information
(
    name VARCHAR PRIMARY KEY,
    phoneNumber VARCHAR,
    whatsapp VARCHAR,
    website VARCHAR,
    instagram VARCHAR,
    discord VARCHAR,
    servicehours VARCHAR
);
